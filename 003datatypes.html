C# DATATYPES

* A data type specifies the size and type of variable values. It is important to use the correct data type for the corresponding variable; to avoid errors, to save time and memory, but it will also make your code more maintainable and readable.

Types of Data Types

sl no Data Type size Description
1 int 4 bytes Stores whole numbers from -2,147,483,648 to 2,147,483,647
2 long 8 bytes Stores whole numbers from -9,223,372,036,854,775,808 to 9,223,372,036,854,775,807
3 float 4 bytes Stores fractional numbers. Sufficient for storing 6 to 7 decimal digits
4 double 8 bytes Stores fractional numbers. Sufficient for storing 15 decimal digits
5 bool 1 bit Stores true or false values
6 char 2 bytes Stores a single character/letter, surrounded by single quotes
7 string 2 bytes per character Stores a sequence of characters, surrounded by double quotes

Different types of Datatypes

Integer (int)

stores integers (whole numbers), without decimals, such as 123 or -123

Correct Syntax

using System;
            
public class Program
  {
     public static void Main()
       {
        int baskets = 16;
        int applesInBasket = 24;
        
        int total = baskets * applesInBasket;
        
        Console.WriteLine("There are total of {total} apples");
        }
  }
//Output There are total of 384 apples

Integer (int)

Wrong Syntax

                                            using System;
            
                                            public class Program
                                              {
                                                 public static void Main()
                                                   {
                                                    integer baskets = 16;//Wrong Declaration of integer
                                                    int applesInBasket = 24 //missing ; (shows error)
                                                    
                                                    int total = baskets * applesInBasket;
                                                    
                                                    Console.WriteLine();//missing variable
                                                    }
                                              }

Different types of Datatypes

Double (D)

stores floating point numbers, with decimals, such as 19.99 or -19.99

Correct Syntax

 using System;
            
 public class Program
     {
        public static void Main()
          {
            double n1 = 0.1 + 0.1 + 0.1;
            double n2 = 1 / 3.0;
            
            if (n1 == n2)
            {
                Console.WriteLine("Numbers are equal");
            }
            else
            {
                Console.WriteLine("Numbers are not equal");
            }
            
           }
      }
//Output Numbers are not equal

Double (D)

Wrong Syntax

using System;
            
 public class Program
     {
        public static void Main()
          {
            d n1 = 0.1 + 0.1 + 0.1;//wrong declaration of double
            double n2 = 1 / 3.0 //missing ; (shows error)
            
            if (n1 == n2)
            {
                Console.WriteLine("Numbers are equal");
            }
            else
            {
                Console.WriteLine("Numbers are not equal");
            }

           }
      }

Different types of Datatypes

Character (char)

stores single characters, such as 'a' or 'B'. Char values are surrounded by single quotes

Correct Syntax

using System;
            
 public class Program
  {
    public static void Main()
     {
        char myGrade;//Declaration of Char
        myGrade = 'B';//Assignment value to the Char myGrade
        Console.WriteLine(myGrade);//Accessing Char.
     }
   }
//Output B

Character (char)

Wrong Syntax

using System;
            
public class Program
   {
      public static void Main()
         {
            character myGrade;//Wrong Declaration of Char
            myGrade = "Axe"//Wrong Assignment value to the Char myGrade
            Console.WriteLine(Axe);//Wrong Accessing Char.
         }
   }

Different types of Datatypes

String (string)

stores text, such as "Hello World". String values are surrounded by double quotes 


using System;
            
public class Program
 {
   public static void Main()
      {
         string myName;//Declaration of string
         myName = "AMAL";//Assignment value to the string myName
         Console.WriteLine(myName);//Accessing string.
     }
  }
//Output JOHN

String (string)

Wrong Syntax

using System;
            
public class Program
   {
      public static void Main()
         {
          str myName;//Wrong Declaration of string
          myName = AMAL;//Wrong Assignment value to the string myName
          myName = 'AMAL'//Wrong Assignment value to the string myName
          Console.WriteLine("AMAL");//Wrong Accessing string.
         }
   }

Different types of Datatypes

Boolean (bool)

stores values with two states: true or false 


using System;

namespace MyApplication
   {
      class Program
        {
            static void Main(string[] args)
              {
                    bool isCSharpFun = true;
                    bool isFishTasty = false;
                    Console.WriteLine(isCSharpFun);   // Outputs True
                    Console.WriteLine(isFishTasty);   // Outputs False    
              }
        }
    }
Wrong Syntax

using System;

namespace MyApplication
    {
      class Program
       {
          static void Main(string[] args)
            {
                  bool isCSharpFun = "amal" //Wrong Declaration of bool
                  bool isFishTasty = 15 //Wrong Declaration of bool
                  Console.WriteLine(true);   // Wrong Accessing bool
                  Console.WriteLine(false);   // Wrong Accessing bool   
            }
        }
    }
